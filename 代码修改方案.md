  Step 5：改造 app/agents/repair_agent.py

  这是改动最大的文件，从单链改为两阶段：

  阶段一：打标链
  self.tagging_llm = ChatOpenAI(model=settings.TAGGING_MODEL_NAME, ...)
  self.tagging_chain = tagging_prompt | self.tagging_llm | keyword_parser

  阶段二：RAG 检索 + 推理链
  async def analyze(self, base64_image: str) -> dict:
      # 1. 打标
      keywords = await self.tagging_chain.ainvoke({"image_url": base64_image})

      # 2. 检索 SOP
      query = " ".join(keywords)
      sop_chunks, is_rag_confident = self.retriever.search(query)
      sop_context = format_sop_context(sop_chunks) if is_rag_confident else ""

      # 3. 推理（注入 SOP 上下文）
      result = await self.chain.ainvoke({
          "image_url": base64_image,
          "sop_context": sop_context
      })

      # 4. 附加路由信息
      dispatch_info = get_dispatch_info(result.category)

      return {**result.model_dump(), "dispatch_info": dispatch_info, "rag_used": is_rag_confident}
